最小生成树（MST）

最小生成树是一个无向图的一个子集，包含所有顶点，且使得边的总权重最小。它可以用于优化连接所有点的成本，例如在网络设计、道路建设和电网布局中。

适用场景

最小生成树适用于以下场景：

  • 网络设计（如计算机网络、电力网络）。
  • 城市交通规划。
  • 运输和物流问题（如最短路径、配送路线）。

时间和空间复杂度

Kruskal 算法：

  • 时间复杂度：O(E log E)，E 为边的数量。
  • 空间复杂度：O(V + E)，V 为顶点数量。

Prim 算法：

  • 时间复杂度：O(E log V)（使用优先队列），O(V²)（使用邻接矩阵）。
  • 空间复杂度：O(V)。

算法思路

最小生成树的基本思路如下：

  • Kruskal 算法：将图的所有边按权重升序排序，然后逐条检查边，选择不形成环的边，直到包含所有顶点。
  • Prim 算法：从任意顶点开始，将其标记为已访问。选择一条权重最小且连接未访问顶点的边，重复直到所有顶点都被访问。

using namespace std;

struct Edge {
    int u, v, weight;
};

bool compare(Edge a, Edge b) {
    return a.weight < b.weight;
}

int find(int parent[], int x) {
    if (parent[x] != x)
        parent[x] = find(parent, parent[x]);
    return parent[x];
}

void unionSet(int parent[], int rank[], int u, int v) {
    int rootU = find(parent, u);
    int rootV = find(parent, v);
    if (rootU != rootV) {
        if (rank[rootU] > rank[rootV]) {
            parent[rootV] = rootU;
        } else {
            parent[rootU] = rootV;
            if (rank[rootU] == rank[rootV])
                rank[rootV]++;
        }
    }
}

int main() {
    int V, E;
    cin >> V >> E;
    vector edges(E);
    for (int i = 0; i < E; i++) {
        cin >> edges[i].u >> edges[i].v >> edges[i].weight;
    }
    
    sort(edges.begin(), edges.end(), compare);
    int parent[V], rank[V] = {0};
    for (int i = 0; i < V; i++) {
        parent[i] = i;
    }

    vector mst;
    for (Edge edge : edges) {
        if (find(parent, edge.u) != find(parent, edge.v)) {
            unionSet(parent, rank, edge.u, edge.v);
            mst.push_back(edge);
        }
    }

    // 输出最小生成树
    for (Edge edge : mst) {
        cout << edge.u << " -- " << edge.v << " = " << edge.weight << endl;
    }
}

    pq.push({0, 0}); // 权重, 顶点

    while (!pq.empty()) {
        int u = pq.top().second;
        pq.pop();
        inMST[u] = true;

        for (auto &edge : graph[u]) {
            int v = edge.first, weight = edge.second;
            if (!inMST[v] && weight < minWeight[v]) {
                minWeight[v] = weight;
                pq.push({weight, v});
            }
        }
    }

    // 输出最小生成树的边权重
    for (int i = 1; i < V; i++) {
        cout << "0 -- " << i << " = " << minWeight[i] << endl;
